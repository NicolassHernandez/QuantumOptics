\section{Squeezed states}
Lets define the squeeze operator 
\begin{align*}
    \text{Squeezing state}\qquad\ket{\xi,0}=\hS(\xi)\ket{0},\quad\text{with}\quad\hS(\xi)=e^{\frac{1}{2}(\xi^*\ha^2-\xi\ha^{\dagger2})},\quad\xi\in\mathbb{C}.
\end{align*}
Lets use it to compute the expectation value for the quadrature operators.
\begin{align*}
    \braket{\xi,0|\hX_1|\xi,0}&=\braket{0|\hS^\dagger(\xi)\left(\frac{\ha+\ha^\dagger}{2}\right)\hS(\xi)|0}=\frac{1}{2}\braket{0|(\hS^\dagger\ha\hS+\hS^\dagger\ha^\dagger\hS)|0}.
\end{align*}
We define displaced ladder operators as
\begin{align*}
    \tilde{a}=\hS^\dagger(\xi)\ha\hS(\xi),\quad\xi=re^{i\phi}.
\end{align*}
Using the BCH formula 
\begin{align*}
    e^LAe^{-L}=A+[L,A]+\frac{1}{2!}[L,[L,A]]+\cdots
\end{align*}
and $A=\ha$ with $L=-\frac{1}{2}(\xi^*\ha^2-\xi\ha^{\dagger2})$ allow us to express the squeezing ladder operator 
\begin{align}
    \text{Squeezing ladder operators}\qquad
    \highlight{
    \begin{array}{l}
        \tilde{a}=\hS^\dagger(\xi)\ha\hS(\xi)=\ha\cosh(r)-\ha^\dagger e^{i\phi}\sinh(r)\\
        \tilde{a}^\dagger=S(\xi)^\dagger\ha^\dagger\hS(\xi)=\ha^\dagger\cosh(r)-\ha e^{i\phi}\sinh(r).
    \end{array}}
\end{align}
We have used several orders. Its derivative wrt $r$ is:
\begin{align*}
    \frac{d\ha}{dr}=\ha\sinh(r)-\ha^\dagger e^{i\phi}\cosh(r)=-e^{i\phi}\tilde{a}^\dagger,\quad\frac{d\tilde{a}^\dagger}{dr}=-e^{-i\phi}\tilde{a}.
\end{align*} 
Taking another derivative yields:
\begin{align*}
    \frac{d^2\tilde{a}}{dr^2}=\tilde{a}.
\end{align*}
Now we can continue the expectation of the quadrature:
\begin{align*}
    \braket{\hX_1}&=\frac{1}{2}\braket{0|(\hS^\dagger\ha\hS+\hS^\dagger\ha^\dagger\hS)|0}=\frac{1}{2}\braket{0|\left[\tilde{a}+\tilde{a}^\dagger\right]|0}\\
    &=\frac{1}{2}\braket{0|[\ha\cosh(r)-e^{i\phi}\ha^\dagger\sinh(r)+\ha^\dagger\cosh(r)-e^{-i\phi}\sinh(r)]|0}=0.\\
    \braket{\hX_2}&=\text{similarly}=0.
\end{align*}
We can do the same for the operator square:
\begin{align*}
    \braket{\hX_1^2}&=\frac{1}{4}\braket{0|\hS^\dagger(\xi)[(\ha+\ha^\dagger)(\ha+\ha^\dagger)]\hS(\xi)|0}=\frac{1}{4}\braket{0|\hS^\dagger(\ha+\ha^\dagger)\hS\hS^\dagger(\ha+\ha^\dagger)\hS|0}\\
    &=\frac{1}{4}\braket{0|(\tilde{a}+\tilde{a}^\dagger)(\tilde{a}+\tilde{a}^\dagger)|0}=\frac{1}{4}\braket{0|e^{-2r}(\ha+\ha^\dagger)^2|0}=\frac{e^{-2r}}{4}.
\end{align*}
Similarly we can obtain the respective quantities for $\hX_2$ and the uncertainties are:
\begin{align*}
    \braket{\hX_1}=\braket{\hX_2}=0 \Longrightarrow
    \Delta\hX_1=\frac{e^{-r}}{2},\quad\Delta\hX_2=\frac{e^r}{2}.
\end{align*}

\subsection{Displaced squeezing state}
Now,
\begin{align*}
    \shortstack{Displaced squeezing state}\qquad\ket{\xi,\alpha}=\hD(\alpha)\hS(\xi)\ket{0}.
\end{align*}
Now, the operators are:
\begin{align}
    \shortstack{Displaced squeezing\\operator}\qquad\highlight{
    \begin{array}{l}
        \tilde{a}=\hS^\dagger(\xi)\hD^\dagger(\alpha)\ha\hD(\alpha)\hS(\xi)=\cosh(r)\ha-e^{i\phi}\sinh(r)\ha^\dagger+\alpha,\\
        \tilde{a}^\dagger=\hS^\dagger(\xi)\hD^\dagger(\alpha)\ha^\dagger\hD(\alpha)\hS(\xi)=\cosh(r)\ha^\dagger-e^{-i\phi}\sinh(r)\ha+\alpha^*.
    \end{array}}
\end{align}
The expected number of photons in a displaced squeezed state is:
\begin{align*}
    \braket{0|\tilde{a}^\dagger\tilde{a}|0}=\braket{0|[\cosh(r)\ha^\dagger-e^{-i\phi}\sinh(r)\ha+\alpha^*][\cosh(r)\ha-e^{i\phi}\sinh(r)\ha^\dagger+\alpha]|0}=\sinh^2|\xi|+|\alpha|^2.
\end{align*}
The expected value in the quadrature value is:
\begin{align*}
    \braket{\xi,\alpha|\hX_1|\xi,\alpha}&=\braket{0|\hS^\dagger\hD^\dagger\hX_1\hD\hS|0}=\braket{0|\hS^\dagger\hD^\dagger(\frac{\ha+\ha^\dagger}{2})\hD\hS|0}=
    \frac{1}{2}\braket{0|(\tilde{a}+\tilde{a}^\dagger)|0}\\
    &=\frac{1}{2}\braket{0|[\cosh(r)\ha-e^{i\phi}\sinh(r)\ha^\dagger+\alpha+\cosh(r)\ha^\dagger-e^{-i\phi}\sinh(r)\ha+\alpha^*]|0}=\re{\alpha}.\\
    \braket{\xi,\alpha|\hX_2|\xi,\alpha}&=\im{\alpha}.
\end{align*}

On the other hand, the quadrature uncertainties are:
\begin{align*}
    \Delta\hX_1^2&=\braket{\xi,\alpha|\hX_1^2|\xi,\alpha}-\braket{\xi,\alpha|\hX_1|\xi,\alpha}^2=\braket{0|\hS^\dagger\hD^\dagger\hX_1\hD\hS\hS^\dagger\hD^\dagger\hX_1\hD\hS|0}-\re{\alpha}^2\\
    &=\braket{0|\left(\frac{\tilde{a}+\tilde{a}^\dagger}{2}\right)\left(\frac{\tilde{a}+\tilde{a}^\dagger}{2}\right)|0}-\re{\alpha}^2\\
    &=\frac{1}{4}\braket{0|[\cosh(r)\ha-e^{i\phi}\sinh(r)\ha^\dagger+\alpha+\cosh(r)\ha^\dagger-e^{-i\phi}\sinh(r)\ha+\alpha^*]^2|0}-\re{\alpha}^2\\
    &=\frac{1}{4}(\cosh^2(r)+\sinh^2(r)-2\cosh(r)\sinh(r)\cos\phi)+\re{\alpha}^2-\re{\alpha}^2\\
    \Delta\hX_1^2&=\frac{1}{4}[\cosh^2(r)+\sinh^2(r)-2\cosh(r)\sinh(r)\cos(\phi)].\\
    \Delta\hX_2^2&=\frac{1}{4}[\cosh^2(r)+\sinh^2(r)+2\cosh(r)\sinh(r)\cos(\phi)].
\end{align*}

\subsection{Displaced squeezed temporal state}
This is the most general case of the two above.

The expected value of $\hX_1(t)$ is 
\begin{align*}
\braket{\hX_1(t)}&=\braket{0|\hS^\dagger\hD^\dagger\frac{\ha e^{-i\omega t}+\ha^\dagger e^{i\omega t}}{2}\hD\hS|0}=\frac{1}{2}\braket{0|\tilde{a}e^{-i\omega t}+\tilde{a}^\dagger e^{i\omega t}|0}\\
&=\frac{1}{2}\bra{0}\left[\cosh(r)\ha e^{-i\omega t}-e^{i\phi}\sinh(r)\ha^\dagger e^{-i\omega t}+\alpha e^{-i\omega t}+\cosh(r)\ha^\dagger e^{i\omega t}
-e^{-i\phi}\sinh(r)\ha e^{i\omega t}+\alpha^*e^{i\omega t}\right]\ket{0}\\
&=\frac{1}{2}[\alpha e^{-i\omega t}+\alpha^*e^{i\omega t}]\\
\braket{\hX_1(t)}&=\re{\alpha e^{-i\omega t}}.
\end{align*}
Similarly,
\begin{align*}
\braket{\hX_2(t)}&=\braket{0|\hS^\dagger\hD^\dagger\frac{\ha e^{-i\omega t}-\ha^\dagger e^{i\omega t}}{2i}\hD\hS|0}=\frac{1}{2i}\braket{0|\tilde{a}e^{-i\omega t}+\tilde{a}^\dagger e^{i\omega t}|0}\\
&=\frac{1}{2i}\bra{0}\left[\cosh(r)\ha e^{-i\omega t}-e^{i\phi}\sinh(r)\ha^\dagger e^{-i\omega t}+\alpha e^{-i\omega t}-\cosh(r)\ha^\dagger e^{i\omega t}
+e^{-i\phi}\sinh(r)\ha e^{i\omega t}-\alpha^*e^{i\omega t}\right]\ket{0}\\
&=\frac{1}{2}[\alpha e^{-i\omega t}-\alpha^*e^{i\omega t}]\\
\braket{\hX_2(t)}&=\im{\alpha e^{-i\omega t}}.
\end{align*}
Doing the same for the operators squared:
\begin{align*}
\braket{\hX_1^2(t)}&=\braket{0|\hS^\dagger\hD^\dagger\left(\frac{\ha e^{-i\omega t}+\ha^\dagger e^{i\omega t}}{2}\right)^2\hD\hS|0}\\
&=\frac{1}{4}\braket{0|\hS^\dagger\hD^\dagger\left(\ha e^{-i\omega t}+\ha^\dagger e^{i\omega t}\right)\hD\hS\hS^\dagger\hD^\dagger\left(\ha e^{-i\omega t}+\ha^\dagger e^{i\omega t}\right)\hD\hS|0}\\
&=\frac{1}{4}\braket{0|(\tilde{a}e^{-i\omega t}+\tilde{a}^\dagger e^{i\omega t})^2|0}\\
&=\frac{1}{4}\left[e^{-i2\omega t}\braket{0|\tilde{a}^2|0}+e^{i2\omega t}\braket{0|\tilde{a}^{\dagger2}|0}+\braket{0|\tilde{a}\tilde{a}^\dagger|0}+\braket{0|\tilde{a}^\dagger\tilde{a}|0}\right].
\end{align*}
We compute each term separately:
\begin{align*}
\braket{0|\tilde{a}^2|0}=&\braket{0|(\cosh(r)\ha-e^{i\phi}\sinh(r)\ha^\dagger+\alpha)(\cosh(r)\ha-e^{i\phi}\sinh(r)\ha^\dagger+\alpha)|0}\\
=&\cosh^2(r)\braket{\ha^2}-e^{i\phi}\sinh(r)\cosh(r)\braket{\ha\ha^\dagger}+\alpha\cosh(r)\braket{\ha}-e^{i\phi}\sinh(r)\cosh(r)\braket{\ha^\dagger\ha}\\
&+e^{i2\phi}\sinh^2(r)\braket{\ha^{\dagger2}}-\alpha e^{i\phi}\sinh(r)\braket{\ha^\dagger}+\alpha\cosh(r)\braket{\ha}-\alpha e^{i\phi}\sinh(r)\braket{\ha^\dagger}+\alpha^2\\
&=\alpha^2-e^{i\phi}\sinh(r)\cosh(r)\\
\braket{0|\tilde{a}^2|0}&=\alpha^2-\frac{1}{2}e^{i\phi}\sinh(2r).
\end{align*}
\begin{align*}
\braket{0|\tilde{a}^{\dagger2}|0}=&\braket{0|(\cosh(r)\ha^\dagger-e^{-i\phi}\sinh(r)\ha+\alpha^*)(\cosh(r)\ha^\dagger-e^{-i\phi}\sinh(r)\ha+\alpha^*)|0}\\
=&\cosh^2(r)\braket{\ha^{\dagger2}}-e^{-i\phi}\cosh(r)\sinh(r)\braket{\ha^\dagger\ha}+\alpha^*\cosh(r)\braket{\ha^\dagger}-e^{-i\phi}\sinh(r)\cosh(r)\braket{\ha\ha^\dagger}\\
&+e^{-i2\phi}\sinh^2(r)\braket{\ha^2}-\alpha^*e^{-i\phi}\sinh(r)\braket{\ha}+\alpha^*\cosh(r)\braket{\ha^\dagger}-\alpha^*e^{-i\phi}\sinh(r)\braket{\ha^\dagger}+\alpha^{*2}\\
=&\alpha^{*2}-e^{-i\phi}\sinh(r)\cosh(r)\\
\braket{0|\tilde{a}^{\dagger2}|0}=&\alpha^{*2}-\frac{1}{2}e^{-i\phi}\sinh(2r).
\end{align*}
\begin{align*}
\braket{0|\tilde{a}\tilde{a}^\dagger|0}=&\braket{0|(\cosh(r)\ha-e^{i\phi}\sinh(r)\ha^\dagger+\alpha)(\cosh(r)\ha^\dagger-e^{-i\phi}\sinh(r)\ha+\alpha^*)|0}\\
\braket{0|\tilde{a}\tilde{a}^\dagger|0}=&\cosh^2(r)+|\alpha|^2.
\end{align*}
\begin{align*}
\braket{0|\tilde{a}^\dagger\tilde{a}|0}=&\braket{0|(\cosh(r)\ha^\dagger-e^{-i\phi}\sinh(r)\ha+\alpha^*)(\cosh(r)\ha-e^{i\phi}\sinh(r)\ha^\dagger+\alpha)|0}\\
\braket{0|\tilde{a}^\dagger\tilde{a}|0}=&\sinh^2(r)+|\alpha|^2.
\end{align*}
Putting all together:
\begin{align*}
\braket{\hX_1^2(t)}&=\frac{1}{4}\left\{[\alpha^2-\frac{1}{2}e^{i\phi}\sinh(2r)]e^{-i2\omega t}+[\alpha^{*2}-\frac{1}{2}e^{-i\phi}\sinh(2r)]e^{i2\omega t}+
\cosh^2(r)+|\alpha|^2+\sinh^2(r)+|\alpha|^2\right\}\\
&=\frac{1}{4}\left[(\alpha e^{-i\omega t}+\alpha^*e^{i\omega t})^2+\cosh(2r)-\sinh(2r)\cos(\phi-2\omega t)\right]\\
\braket{\hX_1^2(t)}&=[\re{\alpha e^{-i\omega t}}]^2+\frac{1}{4}[\cosh(2r)-\sinh(2r)\cos(\phi-2\omega t)].
\end{align*}
The uncertainty is:
\begin{align*}
\Delta\hX_1(t)=\sqrt{\braket{\hX_1^2(t)}-\braket{\hX_1(t)}^2}=\frac{1}{2}\sqrt{\cosh(2r)-\sinh(2r)\cos(\phi-2\omega t)}.
\end{align*}

Similarly,
\begin{align*}
\braket{\hX_2^2(t)}&=\braket{0|\hS^\dagger\hD^\dagger\left(\frac{\ha e^{-i\omega t}-\ha^\dagger e^{i\omega t}}{2i}\right)^2\hD\hS|0}\\
&=-\frac{1}{4}\braket{0|\hS^\dagger\hD^\dagger(\ha e^{-i\omega t}-\ha^\dagger e^{i\omega t})\hD\hS\hS^\dagger\hD^\dagger(\ha e^{-i\omega t}+\ha^\dagger e^{i\omega t})\hD\hS|0}\\
&=-\frac{1}{4}\braket{0|(\tilde{a}e^{-i\omega t}-\tilde{a}^\dagger e^{i\omega t})^2|0}\\
&=-\frac{1}{4}[e^{-i2\omega t}\braket{0|\tilde{a}^2|0}+e^{i2\omega t}\braket{0|\tilde{a}^{\dagger2}|0}-\braket{0|\tilde{a}\tilde{a}^\dagger|0}-\braket{0|\tilde{a}^\dagger\tilde{a}|0}].
\end{align*}
We use the previous results to get
\begin{align*}
\braket{\hX^2_2(t)}&=-\frac{1}{4}\left\{[\alpha^2-\frac{1}{2}e^{i\phi}\sinh(2r)]e^{-i2\omega t}+[\alpha^{*2}-\frac{1}{2}e^{-i\phi}\sinh(2r)]e^{i2\omega t}-
\cosh^2(r)-|\alpha|^2-\sinh^2(r)-|\alpha|^2\right\}\\
&=-\frac{1}{4}[(\alpha e^{-i\omega t}-\alpha^*e^{i\omega t})^2-\cosh(2r)-\sinh(2r)\cos(\phi-2\omega t)]\\
\braket{\hX_2^2(t)}&=[\im{\alpha e^{-i\omega t}}]^2+\frac{1}{4}[\cosh(2r)+\sinh(2r)\cos(\phi-2\omega t)].
\end{align*}
And the uncertainty is:
\begin{align*}
\Delta\hX_2(t)=\sqrt{\braket{\hX_2^2(t)}-\braket{\hX_2(t)}^2}=\frac{1}{2}\sqrt{\cosh(2r)+\sinh(2r)\cos(\phi-2\omega t)}
\end{align*}
We can verify that $\Delta\hX_1(0)=e^{-r}/2$ and $\Delta\hX_2(0)=e^r/2$ when $\phi=t=0$. The following table sumarizes the most improtant 
results for the most general case from which we can get the squeezed or displaced states.
