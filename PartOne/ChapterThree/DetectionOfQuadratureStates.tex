\section{Detection of quadrature states}
Modes at the beam splitter output are:
\begin{align*}
    \hA_1=\frac{1}{\sqrt{2}}(\ha_1+\ha_2),\quad\hA_2=\frac{1}{\sqrt{2}}(\ha_1-\ha_2).
\end{align*}
Detectors $D_1,D_2$ measure the intensity at each output:
\begin{align}
    \text{Photocurrent}\qquad\highlight{\hat{I}_{k}\propto\braket{\hN_{k}}=\braket{\hA^\dagger_{k}\hA_{k}}}.
\end{align}
\begin{figure}[htbp]
    \centering
    \includegraphics[width=.5\columnwidth]{PartOne/ChapterThree/figures/homodynedetection.png}
    \caption{Balanced homodyne detection.}
\end{figure}

Let us consider the number operator in each output:
\begin{align*}
    \hN_1&=\hA_1^\dagger\hA_1=\left(\frac{\ha_1^\dagger+\ha_2^\dagger}{\sqrt{2}}\right)\left(\frac{\ha_1+\ha_2}{\sqrt{2}}\right)=\frac{1}{2}(\ha_1^\dagger\ha_1+\ha_2^\dagger\ha_2+\ha_1^\dagger\ha_2+\ha_2^\dagger\ha_1),\\
    \hN_2&=\hA_2^\dagger\hA_2=\left(\frac{\ha_1^\dagger-\ha_2^\dagger}{\sqrt{2}}\right)\left(\frac{\ha_1-\ha_2}{\sqrt{2}}\right)=\frac{1}{2}(\ha_1^\dagger\ha_1+\ha_2^\dagger\ha_2-\ha_1^\dagger\ha_2-\ha_2^\dagger\ha_1),\\
    \hN_1-\hN_2&=\ha_1^\dagger\ha_2+\ha_2^\dagger\ha_1.
\end{align*}
Let us consider that the input 2 contains a strong \bfemph{coherent state field} that we refer to as \emph{local oscillator}. Thus the total state over 
which the average is being taken is 
\begin{align}
    \underbrace{\ket{\Psi_1}}_{\text{Input 1}}\underbrace{\ket{\alpha_2}}_{\text{Coherent state input 2}}\Longrightarrow \ha_2\ket{\alpha_2}=\alpha_2\ket{\alpha_2}.
\end{align}
This produces the following irradiances: 
\begin{align*}
    I_1&=\braket{\psi_1\alpha_2|\hA^\dagger_{1}\hA_1|\alpha_2\Psi_1}=\braket{\Psi_1\alpha_2|\frac{\ha^\dagger_1+\ha^\dagger_2}{\sqrt{2}}\frac{\ha_1+\ha_2}{\sqrt{2}}|\alpha_2\Psi_1}=
    \frac{1}{2}\braket{\Psi_1\alpha_2|(\ha^\dagger_1+\alpha_2^*)(\ha_1+\alpha_2)|\alpha_2\Psi_1}\\
    &=\frac{1}{2}\braket{\Psi_1|(\ha^\dagger\ha_1+\ha_1\alpha_2^*+\ha^\dagger_1\alpha_2+|\alpha_2|^2)|\Psi_1},\\
    I_2&=\braket{\psi_1\alpha_2|\hA^\dagger_2\hA_2|\alpha_2\Psi_1}=\braket{\Psi_1\alpha_2|\frac{\ha^\dagger_1-\ha^\dagger_2}{\sqrt{2}}\frac{\ha_1-\ha_2}{\sqrt{2}}|\alpha_2\Psi_1}=
    \frac{1}{2}\braket{\Psi_1\alpha_2|(\ha^\dagger_1-\alpha_2^*)(\ha_1-\alpha_2)|\alpha_2\Psi_1}\\
    &=\frac{1}{2}\braket{\Psi_1|(\ha_1^\dagger\ha_1-\ha_1\alpha_2^*-\ha^\dagger_1\alpha_2+|\alpha_2|^2)|\Psi_1}.
\end{align*}
By taking the difference $I_1-I_2$, we can change the nature of $\alpha_2$ to have different quadratures:
\begin{align*}
    I_1-I_2&\propto\braket{\hN_1-\hN_2}=\braket{\Psi_1|\ha_1\alpha_2^*+\ha_1^\dagger\alpha_2|\Psi_1}=\begin{cases}
        2\alpha_2\braket{\Psi_1|\hX_1|\Psi_1},&\alpha_2=\alpha_2^*,\\
        -2i\alpha_2\braket{\Psi_1|\hX_2|\Psi_1},&\alpha_2=-\alpha_2^*.
    \end{cases}
\end{align*}
In general by changing the phase of input 2 we can measure a general quadrature. Noise in intensity corresponds to noise in specific quadratures.

Say $\alpha_2=|\alpha_2|e^{i\varphi_\alpha}$, then:
{\small
\begin{align*}
    \hN_1-\hN_2=|\alpha_2|(\ha_1e^{-i\varphi_\alpha}+\ha_1^\dagger e^{i\varphi_\alpha})=|\alpha_2|(\ha_1+\ha_1^\dagger)\cos\varphi_\alpha+i|\alpha_2|(\ha_1-\ha_2^\dagger)\sin\varphi_\alpha
    =2|\alpha_2|(\cos\varphi_\alpha\hX_1-i\sin\varphi_\alpha\hX_2).
\end{align*}}
The noise in the difference photocurrent is thus proportional to:
\begin{align*}
    \braket{(\hN_1-\hN_2)^2}-\braket{\hN_1-\hN_2}^2=\textcolor{BrickRed}{\braket{(\alpha_2^*\ha_1+\alpha_2\ha_1^\dagger)^2}}-\braket{(\alpha_2^*\ha_1+\alpha_2\ha_1^\dagger)}^2.
\end{align*}
The term in red can be $\hX_1$ or $\hX_2$ depending on the phase of $\alpha_2$, as stated above.
\begin{itemize}[itemsep=0pt,topsep=0pt,parsep=0pt]
    \item Noise in photocurrent difference corresponds to uncertainty in quadratures.
    \item If a quadrature measurement has uncertainty $\Delta X_\phi^2<1/4$.
\end{itemize}
