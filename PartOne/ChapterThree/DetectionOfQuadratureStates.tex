\section{Detection of quadrature states}
Modes at the beam splitter output are:
\begin{align*}
    \hA_1=\frac{1}{\sqrt{2}}(\ha_1+\ha_2),\quad\hA_2=\frac{1}{\sqrt{2}}(\ha_1-\ha_2).
\end{align*}
The intensity operator is the number of photons detected at each detector:
\begin{align}
    \text{Photocurrent}\qquad\highlight{\hI_{k}=\hN_{k}=\hA^\dagger_{k}\hA_{k}}.
\end{align}
\begin{figure}[htbp]
    \centering
    \includegraphics[width=.5\columnwidth]{PartOne/ChapterThree/figures/homodynedetection.png}
    \caption{Balanced homodyne detection.}
\end{figure}

Let us consider the number operator in each output:
\begin{align*}
    \hI_1&=\hA_1^\dagger\hA_1=\left(\frac{\ha_1^\dagger+\ha_2^\dagger}{\sqrt{2}}\right)\left(\frac{\ha_1+\ha_2}{\sqrt{2}}\right)=\frac{1}{2}(\ha_1^\dagger\ha_1+\ha_2^\dagger\ha_2+\ha_1^\dagger\ha_2+\ha_2^\dagger\ha_1),\\
    \hI_2&=\hA_2^\dagger\hA_2=\left(\frac{\ha_1^\dagger-\ha_2^\dagger}{\sqrt{2}}\right)\left(\frac{\ha_1-\ha_2}{\sqrt{2}}\right)=\frac{1}{2}(\ha_1^\dagger\ha_1+\ha_2^\dagger\ha_2-\ha_1^\dagger\ha_2-\ha_2^\dagger\ha_1),\\
    \hI_1-\hI_2&=\ha_1^\dagger\ha_2+\ha_2^\dagger\ha_1.
\end{align*}
Let us consider that the input 2 contains a strong \bfemph{coherent state field} that we refer to as \emph{local oscillator}. Thus the total state over 
which the average is being taken is 
\begin{align}
    \underbrace{\ket{\Psi_1}}_{\text{Input 1}}\underbrace{\ket{\alpha_2}}_{\text{Coherent state input 2}}\Longrightarrow \ha_2\ket{\alpha_2}=\alpha_2\ket{\alpha_2}.
\end{align}
This produces the following expected measured intensity (epected number of photons): 
\begin{align*}
    \braket{\hI_1}&=\braket{\hN_1}=\braket{\Psi_1\alpha_2|\hA^\dagger_{1}\hA_1|\alpha_2\Psi_1}=\braket{\Psi_1\alpha_2|\frac{\ha^\dagger_1+\ha^\dagger_2}{\sqrt{2}}\frac{\ha_1+\ha_2}{\sqrt{2}}|\alpha_2\Psi_1}=
    \frac{1}{2}\braket{\Psi_1\alpha_2|(\ha^\dagger_1+\alpha_2^*)(\ha_1+\alpha_2)|\alpha_2\Psi_1}\\
    &=\frac{1}{2}\braket{\Psi_1|(\ha^\dagger\ha_1+\ha_1\alpha_2^*+\ha^\dagger_1\alpha_2+|\alpha_2|^2)|\Psi_1},\\
    \braket{\hI_2}&=\braket{\hN_2}=\braket{\Psi_1\alpha_2|\hA^\dagger_2\hA_2|\alpha_2\Psi_1}=\braket{\Psi_1\alpha_2|\frac{\ha^\dagger_1-\ha^\dagger_2}{\sqrt{2}}\frac{\ha_1-\ha_2}{\sqrt{2}}|\alpha_2\Psi_1}=
    \frac{1}{2}\braket{\Psi_1\alpha_2|(\ha^\dagger_1-\alpha_2^*)(\ha_1-\alpha_2)|\alpha_2\Psi_1}\\
    &=\frac{1}{2}\braket{\Psi_1|(\ha_1^\dagger\ha_1-\ha_1\alpha_2^*-\ha^\dagger_1\alpha_2+|\alpha_2|^2)|\Psi_1}.
\end{align*}
By taking the difference $\hI_1-\hI_2$, we can change the nature of $\alpha_2$ to have different quadratures:
\begin{align*}
    \braket{\hI_1-\hI_2}=\braket{\hI_1}-\braket{\hI_2}=\braket{\Psi_1|\ha_1\alpha_2^*+\ha_1^\dagger\alpha_2|\Psi_1}=\begin{cases}
        2\alpha_2\braket{\Psi_1|\hX_1|\Psi_1},&\alpha_2=\alpha_2^*,\\
        -2i\alpha_2\braket{\Psi_1|\hX_2|\Psi_1},&\alpha_2=-\alpha_2^*.
    \end{cases}
\end{align*}
In general by changing the phase of input 2 we can measure a general quadrature. Noise in intensity corresponds to noise in specific quadratures.

Say $\alpha_2=|\alpha_2|e^{i\varphi_\alpha}$, then:
{\small
\begin{align*}
    \hN_1-\hN_2=|\alpha_2|(\ha_1e^{-i\varphi_\alpha}+\ha_1^\dagger e^{i\varphi_\alpha})=|\alpha_2|(\ha_1+\ha_1^\dagger)\cos\varphi_\alpha+i|\alpha_2|(\ha_1-\ha_1^\dagger)\sin\varphi_\alpha
    =2|\alpha_2|(\cos\varphi_\alpha\hX_1-i\sin\varphi_\alpha\hX_2).
\end{align*}}
The noise in the difference photocurrent is thus proportional to:
\begin{align*}
    \braket{(\hN_1-\hN_2)^2}-\braket{\hN_1-\hN_2}^2=\textcolor{BrickRed}{\braket{(\alpha_2^*\ha_1+\alpha_2\ha_1^\dagger)^2}}-\braket{(\alpha_2^*\ha_1+\alpha_2\ha_1^\dagger)}^2.
\end{align*}
The term in red can be $\hX_1$ or $\hX_2$ depending on the phase of $\alpha_2$, as stated above.
\begin{itemize}[itemsep=0pt,topsep=0pt,parsep=0pt]
    \item Noise in photocurrent difference corresponds to uncertainty in quadratures.
    \item If a quadrature measurement has uncertainty $\Delta X_\phi^2<1/4$.
\end{itemize}


\begin{example}{Balanced homodyne deecion of squeezed light}
    Consider a squeezed vacuum field mode incident on a balanced homodyne detector.
    \begin{figure}[htbp]
        \centering
        \includegraphics[width=.3\columnwidth]{PartOne/ChapterThree/figures/ex_homodynedetection.png}
    \end{figure}
    
    The ouput modes in the detectors are given as 
    \begin{align*}
        \hb_1=\frac{1}{\sqrt{2}}(\ha_1+\ha_2),\quad\hb_2=\frac{1}{\sqrt{2}}(\ha_1-\ha_2).
    \end{align*}
    What is the difference $\hN_1-\hN_2$, with $\hN_{i}=\hb^\dagger_i\hb_i$, and the variance $\Delta(\hN_1-\hN_2)$ 
    as a function of $\phi_\alpha$? The phase shifter changes the coherent state input so that $\alpha\to\alpha e^{i\phi_\alpha}$.
    Assume that $|\alpha|\gg1$ and keep terms to linear order in $|\alpha|$.
    \subsubsection{Solution}
    In general, the photon number for output $i$ is $\hN_i=\hat{b}^\dagger_i\hat{b}_i$.
    For each output we have:
    \begin{align*}
    \hN_1&=\frac{1}{2}(\ha_1^\dagger+\ha_2^\dagger)(\ha_1+\ha_2)=\frac{1}{2}(\ha_1^\dagger\ha_1+\ha_1^\dagger\ha_2+\ha_2^\dagger\ha_1+\ha_2^\dagger\ha_2),\\
    \hN_2&=\frac{1}{2}(\ha_1^\dagger-\ha_2^\dagger)(\ha_1-\ha_2)=\frac{1}{2}(\ha_1^\dagger\ha_1-\ha_1^\dagger\ha_2-\ha_2^\dagger\ha_1+\ha_2^\dagger\ha_2).
    \end{align*}
    Substraction gives 
    \begin{align*}
    \hN_1-\hN_2=\frac{1}{2}\left[(\ha_1^\dagger\ha_1+\ha_1^\dagger\ha_2+\ha_2^\dagger\ha_1+\ha_2^\dagger\ha_2)-(\ha_1^\dagger\ha_1-\ha_1^\dagger\ha_2-\ha_2^\dagger\ha_1+\ha_2^\dagger\ha_2)\right]=\ha_1^\dagger\ha_2+\ha_2^\dagger\ha_1.
    \end{align*}

    With the phase shift, the input 2 is:
    \begin{align*}
    \alpha\to\alpha e^{i\phi_\alpha},\quad\text{and}\quad\ha_2\approx\alpha e^{i\phi_\alpha}.
    \end{align*}
    The number photon difference is then:
    \begin{align*}
    \hN_1-\hN_2=\ha_1^\dagger\alpha e^{i\phi_\alpha}+\ha_1\alpha^*e^{-i\phi_\alpha}.
    \end{align*}
    If $\alpha=|\alpha|$, then 
    \begin{align*}
    \hN_1-\hN_2=|\alpha|\left(\ha_1^\dagger e^{i\phi_\alpha}+\ha_1e^{-i\phi_\alpha}\right)=2|\alpha|\hX_{\phi_\alpha},\quad\hX_{\phi_\alpha}=\frac{\ha_1^\dagger e^{i\phi_\alpha}+\ha_1e^{-i\phi_\alpha}}{2},
    \end{align*}
    where $\hX_{\phi_\alpha}$ is the quadrature of mode 1 modulated by the phase shift $\phi_\alpha$.

    To get the uncertainty $\Delta(\hN_1-\hN_2)$, we compute the expected alue of the difference and the expected value of the difference squared.
    We can use the results from the previoous exercise as the input 1 is a squeezed state of the vacuum.
    \begin{align*}
    _1\bra{\xi,0}_2\bra{\alpha}(\hN_1-\hN_2)\ket{\alpha}_2\ket{\xi,0}_1&=2|\alpha|\braket{\hX_{\phi_\alpha}}=0,\\
    _1\bra{\xi,0}_2\bra{\alpha}(\hN_1-\hN_2)^2\ket{\alpha}_2\ket{\xi,0}_1&=4|\alpha|^2\braket{\hX_{\phi_\alpha}^2}=|\alpha|^2[\cosh(2r)-\sinh(2r)\cos(\theta-2\phi_\alpha)],
    \end{align*}
    where $\xi=re^{i\theta}$ is the squeezing factor, and $\phi_\alpha$ the phase shift imposed by the input 2. Uncertainty is therefore:
    \begin{align*}
    \Delta(\hN_1-\hN_2)=|\alpha|\sqrt{\cosh(2r)-\sinh(2r)\cos(\theta-2\phi_\alpha)}.
    \end{align*}
\end{example}
