\section*{Exercise 1}
The Hamiltonian in position space of the quantum harmonic oscillator is:
\begin{align*}
  \hH=\frac{\hp^2}{2m}+\frac{1}{2}m\omega^2\hx^2,\quad[\hx,\hp]=i\hbar.
\end{align*}
Defining the \emph{ladder operators}
\begin{align*}
  \begin{array}{l}
    \ha=\dfrac{1}{\sqrt{2}}\left(\dfrac{\hx}{\sigma}+i\dfrac{\sigma\hp}{\hbar}\right)\\
    \ha^\dagger=\dfrac{1}{\sqrt{2}}\left(\dfrac{\hx}{\sigma}-i\dfrac{\sigma\hp}{\hbar}\right)
  \end{array},\quad\text{with}\quad
  \begin{array}{l}
    \hx=\dfrac{\sigma}{\sqrt{2}}(\ha^\dagger+\ha)\\
    \hp=\dfrac{i\hbar}{\sigma\sqrt{2}}(\ha^\dagger-\ha)
  \end{array},
\end{align*}
with $\sigma$ the oscillation length 
\begin{align*}
  \sigma=\sqrt{\frac{\hbar}{m\omega}},
\end{align*}
allow us to express the Hamiltonian in a simpler form. First, we notice that the commutation of $\hx$ and $\hp$ is inherited for $\ha$ and $\ha^\dagger$:
\begin{align*}
  [\ha,\ha^\dagger]=\frac{1}{2}\left[\frac{\hx}{\sigma}+i\frac{\sigma\hp}{\hbar},\frac{\hx}{\sigma}-i\frac{\sigma\hp}{\hbar}\right]
  =\frac{1}{2}\left(-\frac{2i}{\hbar}[\hx,\hp]\right)
  =-\frac{i}{\hbar}[\hx,\hp]
  =-\frac{i}{\hbar}(i\hbar)
  =1.
\end{align*}
If we replace the $\hx,\hp$ in terms of $\ha,\ha^\dagger$ in the Hamiltonian, we have:
\begin{align*}
  \hH&=\frac{1}{2m}\left[-\frac{\hbar^2}{2\sigma^2}(\ha^\dagger-\ha)^2\right]+\frac{1}{2}m\omega^2\left[\frac{\sigma^2}{2}(\ha^\dagger+\ha)^2\right]\\
  &=\frac{\hbar\omega}{2}\left[\ha^{\dagger2}+\ha^\dagger\ha+\ha\ha^\dagger+\ha^2-\ha^{\dagger2}+\ha^\dagger\ha+\ha\ha^\dagger-\ha^2\right]\\
  &=\frac{\hbar\omega}{2}\left[2(\ha^\dagger\ha+\ha\ha^\dagger)\right]\\
  &=\hbar\omega(\ha^\dagger\ha+\ha^\dagger\ha+1)\\
  &=\hbar\omega\left(\ha^\dagger\ha+\frac{1}{2}\right)\\
  \hH&=\hbar\omega\left(\hN+\frac{1}{2}\right),
\end{align*}
where $\hN=\ha^\dagger\ha$ is the \emph{number operator}. If we know the eigenequation of $\hN$ we automatically know the eigenequation of $\hH$. 
Assuming the following eigenequation for $\hN$, then:
\begin{align*}
  \hN\ket{n}=n\ket{n}\Longrightarrow\hH\ket{n}=\left(n+\frac{1}{2}\right)\ket{n}.
\end{align*}
This means that the energy engenvalues are 
\begin{align*}
  E_n=\left(n+\frac{1}{2}\right)\hbar\omega.
\end{align*}
The nature of $n$ needs to be proved. First, we compute the commutator of $\hN$ with $\ha,\ha^\dagger$:
\begin{align*}
  [\hN,\ha]&=[\ha^\dagger\ha,\ha]=\ha^\dagger[\ha,\ha]+[\ha^\dagger,\ha]\ha=-\ha,\\
  [\hN,\ha^\dagger]&=[\ha^\dagger\ha,\ha^\dagger]=\ha^\dagger[\ha,\ha^\dagger]+[\ha^\dagger,\ha^\dagger]\ha=\ha^\dagger.
\end{align*}
Using these relations, we can test some actions on $\ket{n}$:
\begin{align*}
  \hN\ha^\dagger\ket{n}&=([\hN,\ha^\dagger]+\ha^\dagger\hN)\ket{n}=(\ha^\dagger+\ha^\dagger\hN)\ket{n}=(\ha^\dagger+\ha^\dagger n)\ket{n}=(n+1)\ha^\dagger\ket{n},\\
  \hN\ha\ket{n}&=([\hN,\ha]+\ha\hN)\ket{n}=(-\ha+\ha\hN)\ket{n}=(\ha n-\ha)\ket{n}=(n-1)\ha\ket{n}.
\end{align*}
This imply that $\ha^\dagger\ket{n}$ and $\ha\ket{n}$ are also eigenkets of $\hN$, with eigenvalue increased and decreased by one, respectively.
With this, we conclude that $n$ must be integer so that $\hN$ and therefore $\hH$ are quantized. A more rigorous proof is given by \cite{cohen1986quantum}.

\section*{Exercise 2}
\begin{enumerate}[itemsep=0pt,topsep=0pt,label=\alph*)]
  \item Ill use the definition of the E- and B-fields in terms of the vector potential.
  \begin{align*}
    \bE=-\partial_t\bA,\quad\text{and}\quad\bB=\nabla\times\bA.
  \end{align*} 
  Taking the curl of the B-field allows to replace the vector potential and have a triple product,
  which can be reexpressed using vector identities. Because $\nabla\times\bB$ is Ampere's law,
  it also depends on the E-field, where the above definition can be replaced.
  \begin{align*}
    \nabla\times\bB&=\frac{1}{c^2}\partial_t\bE\\
    \nabla\times\nabla\times\bA&=\frac{1}{c^2}\partial_t(-\partial_t\bA)\\
    \nabla(\cancelto{0}{\nabla\cdot\bA})-\nabla^2\bA&=-\frac{1}{c^2}\partial_t^2\bA\\
    \nabla^2\bA-\frac{1}{c^2}\partial_t^2\bA&=0.
  \end{align*}
  \item To show that the expression provided is a solution of the A-potential wave equation, we need to substitute it into the wave equation and have 
  an equality. 
  The second time-derivative is:
  \begin{align*}
    \partial_t\bA(z,t)&=\partial_t\left[\sum_{\bk,\lambda}e_{\bk,\lambda}[A_{\bk,\lambda}e^{-i\omega t}+A^*_{\bk,\lambda}e^{i\omega t}]\sin(kz)\right]=i\sum_{\bk,\lambda}\omega e_{\bk,\lambda}[A^*_{\bk,\lambda}e^{i\omega t}-A_{\bk,\lambda}e^{-i\omega t}]\sin(kz),\\
    \partial^2_t\bA(z,t)&=\partial_t\left[i\sum_{\bk,\lambda}\omega e_{\bk,\lambda}[A^*_{\bk,\lambda}e^{i\omega t}-A_{\bk,\lambda}e^{-i\omega t}]\sin(kz)\right]=-\sum_{\bk,\lambda}\omega^2e_{\bk,\lambda}[A^*_{\bk,\lambda}e^{i\omega t}+A_{\bk,\lambda}e^{-i\omega t}]\sin(kz).
  \end{align*}
  For the laplacian, we have:
  \begin{align*}
    \nabla^2\bA(z,t)&=-\sum_{\bk,\lambda}k^2e_{\bk,\lambda}[A_{\bk,\lambda}e^{-i\omega t}+A^*_{\bk,\lambda}e^{i\omega t}]\sin(kz).
  \end{align*}
  Substituting both, and using the dispersion relation $\omega=|\bk|c$:
  {\small
  \begin{align*}
    \nabla^2\bA-\frac{1}{c^2}\partial_t^2\bA&=0\\
    -\sum_{\bk,\lambda}k^2e_{\bk,\lambda}[A_{\bk,\lambda}e^{-i\omega t}+A^*_{\bk,\lambda}e^{i\omega t}]\sin(kz)-\frac{1}{c^2}\left[-\sum_{\bk,\lambda}\omega^2_{\bk}e_{\bk,\lambda}[A^*_{\bk,\lambda}e^{i\omega t}+A_{\bk,\lambda}e^{-i\omega t}]\sin(kz)\right]&=\\
    \sum_{\bk,\lambda}\left[e_{\bk,\lambda}[A_{\bk,\lambda}e^{-i\omega t}+A^*_{\bk,\lambda}e^{i\omega t}]\sin(kz)\left(\frac{\omega^2_{\bk}}{c^2}-k^2\right)\right]&=\\
    \sum_{\bk,\lambda}\left[e_{\bk,\lambda}[A_{\bk,\lambda}e^{-i\omega t}+A^*_{\bk,\lambda}e^{i\omega t}]\sin(kz)\left(k^2-k^2\right)\right]&=\quad(\omega^2=|\bk|^2c^2)\\
    0&=0.
  \end{align*}}
  We could also propose a standard harmonic solution, substitute it into the vector equation and impose boundary conditions. The one I did is easier. Also, I denoted 
  $\omega_{\bk}$ to remember that $\omega$ depends on $\bk$. The dispersion relation holds for each mode $\bk$.
  \item Assuming a x polarized vector potential
  \begin{align*}
    \bA(z,t)=\hx[Ae^{-i\omega t}+A^*e^{i\omega t}]\sin(kz).
  \end{align*}
  The E- and B-fields are:
  \begin{align*}
    \bE(z,t)&=-\partial_t\bA(z,t)=i\omega\hx[Ae^{-i\omega t}-A^*e^{i\omega t}]\sin(kz)\equiv2\omega|A|\hx\sin(\omega t-\phi)\sin(kz),\\
    \bB(z,t)&=\nabla\times\bA(z,t)=\hy\partial_zA_x=\hy k[Ae^{-i\omega t}+A^*e^{i\omega t}]\cos(kz)\equiv2k|A|\hy\cos(\omega t-\phi)\cos(kz),
  \end{align*}
  where we have assumed a phasor form of $A$: $A=|A|e^{i\phi}$.
  \item We use the expression from previous part. We first compute $\bE^2_{\bk,\lambda}$ and $\bB^2_{\bk,\lambda}$
  \begin{align*}
    \bE^2&=4\omega^2|A|^2\sin^2(\omega t-\phi)\sin^2(kz),\\
    \bB^2&=4k^2|A|^2\cos^2(\omega t-\phi)\cos^2(kz).
  \end{align*}
  Putting these two in the volume integral, and recalling that 
  \begin{align*}
    \int_0^L\;dz\sin^2(kz)=\int_0^L\;dz\cos^2(kz)=\frac{L}{2},
  \end{align*}
  yields
  \begin{align*}
    H_\omega&=\int_0^L\;dV\left\{2\varepsilon_0\omega^2|A|^2\sin^2(\omega t-\phi)\sin^2(kz)+2\frac{k^2}{\mu_0}|A|^2\cos^2(\omega t-\phi)\cos^2(kz)\right\}\\
    &=2\varepsilon_0\omega^2|A|^2\sin^2(\omega t-\phi)S\int_0^L\;dz\sin^2(kz)+2\frac{k^2}{\mu_0}|A|^2\cos^2(\omega t-\phi)S\int_0^L\;dz\cos^2(kz)\\
    H_\omega&=V\left[\varepsilon_0\omega^2|A|^2\sin^2(\omega t-\phi)+\frac{k^2}{\mu_0}|A|^2\cos^2(\omega t-\phi)\right].
  \end{align*}
  Using the fact that $\omega=kc$, $1/\mu_0=\varepsilon_0c^2$, then $k^2/\mu_0=(\omega/c)^2(\varepsilon_0c^2)=\varepsilon_0\omega^2$, which is the same as the E-field factor term.
  Therefore,
  \begin{align*}
    H_\omega=V\varepsilon_0\omega^2|A|[\sin^2(\omega t-\phi)+\cos^2(\omega t-\phi)]=\varepsilon_0V\omega^2|A|^2.
  \end{align*}
  \item Equating the last result with $\hbar\omega$ and solving for $A$:
  \begin{align*}
    \varepsilon_0V\omega^2|A|^2&=\hbar\omega\\
    |A|&=\sqrt{\frac{\hbar}{\varepsilon_0 V\omega}}.
  \end{align*}
  Its phase, $\phi$, remains unknown, but it acts as a global phase factor so it has no physical meaning. The E- and B-fields are:
  \begin{align*}
    \bE(z,t)&=2\sqrt{\frac{\hbar\omega}{\varepsilon_0V}}\hx\sin(\omega t-\phi)\sin(kz),\\
    \bB(z,t)&=\frac{2}{c}\sqrt{\frac{\hbar\omega}{\varepsilon_0V}}\hy\cos(\omega t-\phi)\cos(kz).
  \end{align*}
\end{enumerate}

\section*{Exercise 3}
\begin{enumerate}[itemsep=0pt,topsep=0pt,label=\alph*)]
  \item We can simplify the problem using the BCH formula
  \begin{align*}
    e^{\hA}\hB e^{-\hA}=\hB+[\hA,\hB]+\frac{1}{2!}[\hA,[\hA,\hB]]+\cdots
  \end{align*}
  In this case, $\hA=\alpha^*\ha-\alpha\ha^\dagger$ and $\hB=\ha$. The first two commutators are:
  \begin{align*}
    [\hA,\hB]&=[(\alpha^*\ha-\alpha\ha^\dagger),\ha]=\alpha^*[\ha,\ha]-\alpha[\ha^\dagger,\ha]=\alpha,\\
    [\hA,[\hA,\hB]]&=[(\alpha^*\ha-\alpha\ha^\dagger),\alpha]=0,\\
    &\vdots
  \end{align*}
  Now, we can express the displacement operator as:
  \begin{align*}
    \tilde{a}&=\D^\dagger(\alpha)\ha\D(\alpha)=e^{\alpha^*\ha-\alpha\ha^\dagger}\ha e^{-(\alpha^*\ha-\alpha\ha^\dagger)}=\ha+\alpha,\quad\text{and}\\
    \tilde{a}^\dagger&=\D^\dagger(\alpha)\ha^\dagger\D(\alpha)=\ha^\dagger+\alpha^*.
  \end{align*}
  Application of both in $\ket{0}$ (just to test them) yields:
  \begin{align*}
    \tilde{a}\ket{0}=(\ha+\alpha)\ket{0}=\alpha\ket{0},\quad\text{and}\quad\tilde{a}^\dagger\ket{0}=(\ha^\dagger+\alpha^*)\ket{0}=\ket{1}+\alpha^*\ket{0}.
  \end{align*}
  \item We use algabra of operators and the fact that the coherent state $\ket{\alpha}$ is got from $\ket{0}$ by appliying a displacement operator $\D(\alpha)$:
  \begin{align*}
    \braket{\hat{X}_1}&=\frac{1}{2}\braket{\alpha|(\ha+\ha^\dagger)|\alpha}\\
    &=\frac{1}{2}\braket{0|\D^\dagger(\alpha)(\ha+\ha^\dagger)\D(\alpha)|0}\\
    &=\frac{1}{2}\left[\braket{0|\D^\dagger(\alpha)\ha\D(\alpha)|0}+\braket{0|\D^\dagger(\alpha)\ha^\dagger\D(\alpha)|0}\right]\\
    &=\frac{1}{2}\left[\braket{0|\tilde{a}|0}+\braket{0|\tilde{a}^\dagger|0}\right]\\
    &=\frac{1}{2}\left[\braket{0|(\ha+\alpha)|0}+\braket{0|(\ha^\dagger+\alpha^*)|0}\right]\\
    &=\frac{1}{2}\left[\braket{0|\ha|0}+\alpha+\braket{0|\ha^\dagger|0}+\alpha^*\right]\\
    &=\frac{1}{2}\left[\alpha+\alpha^*\right]\\
    \braket{\hat{X}_1}&=\re{\alpha}.
  \end{align*}
  For $\hat{X}_2$, and using the developmente from above:
  \begin{align*}
    \braket{\hat{X}_2}&=\frac{1}{2i}\left[\braket{0|(\ha+\alpha)|0}-\braket{0|(\ha^\dagger+\alpha^*)|0}\right]\\
    &=\frac{1}{2i}\left[\braket{0|\ha|0}+\alpha-\braket{0|\ha^\dagger|0}-\alpha^*\right]\\
    &=\frac{1}{2i}\left[\alpha-\alpha^*\right]\\
    \braket{\hat{X}_2}&=\im{\alpha}.
  \end{align*}
  For the uncertainties, we now need to get the mean value of $\hat{X}_1^2$ and $\hat{X}_2^2$, with the analogous process:
  \begin{align*}
    \braket{\hat{X}_1^2}=&\frac{1}{4}\braket{0|\D^\dagger(\alpha)[\ha+\ha^\dagger]^2\D(\alpha)|0}\\
    =&\frac{1}{4}\bra{0}\left[\D^\dagger(\alpha)[\ha^2+\ha\ha^\dagger+\ha^\dagger\ha+\ha^{\dagger2}]\D(\alpha)\right]\ket{0}\\
    =&\frac{1}{4}\bra{0}\left[\D^\dagger\ha^2\D(\alpha)+2\D^\dagger(\alpha)\ha^\dagger\ha\D(\alpha)+1+\D^\dagger(\alpha)\ha^{\dagger2}\D(\alpha)\right]\ket{0}\\
    =&\frac{1}{4}\bra{0}\left[(\D^\dagger\ha\D)(\D^\dagger\ha\D)+2(\D^\dagger\ha^\dagger\D)(\D^\dagger\ha\D)+1+(\D^\dagger\ha^\dagger\D)(\D^\dagger\ha^\dagger\D)\right]\ket{0}\\
    =&\frac{1}{4}\bra{0}\left[(\ha+\alpha)^2+2(\ha^\dagger+\alpha^*)(\ha+\alpha)+1+(\ha^\dagger+\alpha^*)^2\right]\ket{0}\\
    =&\frac{1}{4}[\ha^2+2\alpha\ha+\alpha^2+2\ha^\dagger\ha+2\ha^\dagger\alpha+2\alpha^*\ha+2|\alpha|^2+1+\ha^{\dagger2}+2\alpha^*\ha^\dagger+\alpha^{*2}]\\
    \braket{\hat{X}_1^2}=&\frac{1}{4}\left[(\alpha+\alpha^*)^2+1\right].
  \end{align*}
  Then,
  \begin{align*}
    \braket{\Delta\hat{X}_1}=\sqrt{\braket{\hat{X}^2_1}-\braket{\hat{X}_1}^2}
    =\sqrt{\frac{1}{4}\left[(\alpha+\alpha^*)^2+1\right]-\frac{1}{4}(\alpha+\alpha^*)^2}
    =\frac{1}{2}.
  \end{align*}
  The same procedure is done for the uncertainty of $\hat{X}_2$:
  \begin{align*}
    \braket{\hat{X}_2}&=\frac{1}{4}\bra{0}\left[(\D^\dagger\ha\D)(\D^\dagger\ha\D)-2(\D^\dagger\ha^\dagger\D)(\D^\dagger\ha\D)-1+(\D^\dagger\ha^\dagger\D)(\D^\dagger\ha^\dagger\D)\right]\ket{0}\\
    &=-\frac{1}{4}\bra{0}\left[(\ha+\alpha)^2-2(\ha^\dagger+\alpha^*)(\ha+\alpha)-1+(\ha^\dagger+\alpha^*)^2\right]\ket{0}\\
    &=-\frac{1}{4}[\ha^2+2\alpha\ha+\alpha^2-2\ha^\dagger\ha-2\ha^\dagger\alpha-2\alpha^*\ha-2|\alpha|^2-1+\ha^{\dagger2}+2\alpha^*\ha^\dagger+\alpha^{*2}]\\
    \braket{\hat{X}_2^2}&=\frac{1}{4}\left[1-(\alpha-\alpha^*)^2\right].
  \end{align*}
  Then,
  \begin{align*}
    \braket{\Delta\hat{X}_2}=\sqrt{\braket{\hat{X}^2_2}-\braket{\hat{X}_2}^2}=\sqrt{\frac{1}{4}[1-(\alpha-\alpha^*)^2]+\frac{1}{4}(\alpha-\alpha^*)^2}
    =\frac{1}{2}.
  \end{align*}
\end{enumerate}
